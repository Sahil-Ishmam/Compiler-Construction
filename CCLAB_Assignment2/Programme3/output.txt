int -> INT
a -> ID
= -> ASSIGN
1 -> NUMBER
,b -> ID
= -> ASSIGN
0 -> NUMBER
; -> SEMI
printf -> ID
( -> LP
"a -> ID
&&b -> ID
:%d -> ID
\n -> ID
",a -> ID
&&b -> ID
) -> RP
; -> SEMI
printf -> ID
( -> LP
"a -> ID
||b -> ID
:%d -> ID
\n -> ID
",a -> ID
||b -> ID
) -> RP
; -> SEMI
printf -> ID
( -> LP
"!a -> ID
:%d -> ID
\n -> ID
",!a -> ID
) -> RP
; -> SEMI
printf -> ID
( -> LP
"!b -> ID
:%d -> ID
\n -> ID
",!b -> ID
) -> RP
; -> SEMI
